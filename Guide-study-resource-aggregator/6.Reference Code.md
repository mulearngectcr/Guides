## Complete Code

```python
import os
import requests
from dotenv import load_dotenv

load_dotenv()

YOUTUBE_API_KEY = os.getenv("YOUTUBE_API_KEY")
OPENROUTER_API_KEY = os.getenv("OPENROUTER_API_KEY")


def fetch_books(topic):
    url = f"https://openlibrary.org/search.json?q={topic}"

    response = requests.get(url)

    data = response.json()

    books = []

    for book in data["docs"][:5]:
        books.append(
            {
                "title": book.get("title"),
                "author": book.get(
                    "author_name",
                    ["Unknown"]
                )[0],
            }
        )

    return books


def fetch_youtube_videos(topic):

    url = (
        "https://www.googleapis.com/"
        "youtube/v3/search"
    )

    params = {
        "part": "snippet",
        "q": topic,
        "type": "video",
        "maxResults": 5,
        "key": YOUTUBE_API_KEY,
    }

    response = requests.get(
        url,
        params=params,
    )

    videos = []

    for item in response.json()["items"]:

        videos.append(
            {
                "title": item["snippet"]["title"],
                "channel": item["snippet"][
                    "channelTitle"
                ],
                "url": (
                    "https://youtube.com/watch?v="
                    + item["id"]["videoId"]
                ),
            }
        )

    return videos


def build_prompt(
    topic,
    days,
    videos,
    books,
):

    video_text = "\n".join(
        [
            f"- {video['title']} ({video['url']})"
            for video in videos
        ]
    )

    book_text = "\n".join(
        [
            f"- {book['title']} by {book['author']}"
            for book in books
        ]
    )

    prompt = f"""
You are a learning assistant.

Topic: {topic}

Total Days: {days}

Videos:
{video_text}

Books:
{book_text}

Task:

1. Create a day-wise roadmap.
2. Distribute learning evenly across the days.
3. Mention what should be studied each day.
4. Recommend videos when relevant.
5. Recommend books when relevant.
6. Keep the roadmap beginner-friendly.

Return only the roadmap.
"""

    return prompt


def generate_roadmap(prompt):

    url = (
        "https://openrouter.ai/api/v1/"
        "chat/completions"
    )

    headers = {
        "Authorization": (
            f"Bearer {OPENROUTER_API_KEY}"
        ),
        "Content-Type": "application/json",
    }

    payload = {
        "model": "openai/gpt-oss-20b",
        "messages": [
            {
                "role": "user",
                "content": prompt,
            }
        ],
    }

    response = requests.post(
        url,
        headers=headers,
        json=payload,
    )

    return (
        response.json()["choices"][0]
        ["message"]["content"]
    )


def main():

    print(
        "Study Resource Aggregator "
        "+ AI Roadmap Generator"
    )

    topic = input(
        "\nEnter topic: "
    )

    days = input(
        "Enter number of days: "
    )

    print(
        "\nFetching resources..."
    )

    books = fetch_books(topic)

    videos = fetch_youtube_videos(
        topic
    )

    print(
        "Generating roadmap...\n"
    )

    prompt = build_prompt(
        topic,
        days,
        videos,
        books,
    )

    roadmap = generate_roadmap(
        prompt
    )

    print(
        "\nGenerated Roadmap:\n"
    )

    print(roadmap)


if __name__ == "__main__":
    main()
```
